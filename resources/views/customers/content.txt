var villageSelect = new TomSelect('#village_id', {

    if (!query.length) return callback();

    url: '/api/villages/search',
    type: 'get',
    data: {
        q: query
    },
    success: function(res) {
        callback(res);
    }
    });
},
render: {
    option: function(item, escape) {
    return '<div>' + escape(item.name_en) + ' - ' + escape(item.name_kh) + '</div>';
    }
}
});

// Form Submit
$('#customerForm').on('submit', function(e) {
e.preventDefault();

var formData = $(this).serialize();
var customerId = $('#cust_id').val();
var url = customerId ? `/customers/${customerId}` : '/customers';
var method = customerId ? 'PUT' : 'POST';

$.ajax({
    url: url,
    type: method,
    data: formData,
    headers: {
    'X-CSRF-TOKEN': $('meta[name="csrf-token"]').attr('content')
    },
    success: function(response) {
    if (response.success) {
        $('#customerModal').modal('hide');
        toastr.success(response.message || translations.operationCompleted);
        table.ajax.reload();
    }
    },
    error: function(xhr) {
    if (xhr.responseJSON) {
        if (xhr.responseJSON.errors) {
        $.each(xhr.responseJSON.errors, function(key, value) {
            toastr.error(value[0]);
        });
        } else if (xhr.responseJSON.message) {
        toastr.error(xhr.responseJSON.message);
        } else {
        toastr.error(translations.errorOccurred);
        }
    } else {
        toastr.error(translations.errorOccurred);
    }
    }
});
});

// Edit Button
$(document).on('click', '.edit-btn', function() {
var customerId = $(this).data('id');

$.ajax({
    url: `/api/customers/${customerId}`,
    type: 'GET',
    dataType: 'json',
    success: function(customer) {
    console.log('Customer data:', customer);

    // Set form title first
    $('#modalTitle').text(translations.editCustomer);

    // Set all basic fields
    $('#cust_id').val(customer.cust_id || '');
    $('#ic_no').val(customer.ic_no || '');
    $('#id_no').val(customer.id_no || '');
    $('#name_en').val(customer.name_en || '');
    $('#name_kh').val(customer.name_kh || '');
    $('#gender').val(customer.gender || '');
    $('#marital_status').val(customer.marital_status !== null && customer.marital_status !==
        undefined ? customer.marital_status : '');
    // Ensure marital status is properly selected
    setTimeout(function() {
        $('#marital_status').val(customer.marital_status !== null && customer.marital_status !==
        undefined ? customer.marital_status : '');
    }, 50);
    $('#dob').val(customer.dob || '');
    $('#phone1').val(customer.phone1 || '');
    $('#phone2').val(customer.phone2 || '');
    $('#phone3').val(customer.phone3 || '');
    $('#email').val(customer.email || '');
    $('#pob').val(customer.pob || '');
    $('#occupation').val(customer.occupation || '');
    $('#address').val(customer.address || '');
    $('#remark').val(customer.remark || '');

    // Spouse fields
    $('#spouse_en').val(customer.spouse_en || '');
    $('#spouse_kh').val(customer.spouse_kh || '');
    $('#spouse_dob').val(customer.spouse_dob || '');

    // Clear and set village dropdown
    villageSelect.clear();
    villageSelect.clearOptions();
    if (customer.village_id && customer.village) {
        villageSelect.addOption({
        id: customer.village_id,
        name_en: customer.village.name_en,
        name_kh: customer.village.name_kh
        });
        setTimeout(function() {
        villageSelect.setValue(customer.village_id);
        }, 100);
    }

    // Clear and set nationality dropdown
    nationalitySelect.clear();
    if (customer.nationality_id) {
        setTimeout(function() {
        nationalitySelect.setValue(customer.nationality_id);
        }, 100);
    }

    // Clear and set staff dropdown
    staffSelect.clear();
    if (customer.staff_id) {
        setTimeout(function() {
        staffSelect.setValue(customer.staff_id);
        }, 100);
    }

    // Show modal after all data is set
    $('#customerModal').modal('show');
    },
    error: function(xhr, status, error) {
    console.error('Error loading customer:', xhr);
    console.error('Status:', status);
    console.error('Error:', error);
    console.error('Response:', xhr.responseText);
    console.error('Status Code:', xhr.status);
    toastr.error(translations.failedLoadCustomer + ': ' + (xhr.status || translations
        .unknownError));
    }
});
});

// Delete Button
$(document).on('click', '.delete-btn', function() {
var customerId = $(this).data('id');

Swal.fire({
    title: translations.areYouSure,
    text: translations.cannotRevert,
    icon: 'warning',
    showCancelButton: true,
    confirmButtonColor: '#d33',
    cancelButtonColor: '#3085d6',
    confirmButtonText: translations.yesDelete
}).then((result) => {
    if (result.isConfirmed) {
    $.ajax({
        url: `/customers/${customerId}`,
        type: 'DELETE',
        headers: {
        'X-CSRF-TOKEN': $('meta[name="csrf-token"]').attr('content')
        },
        success: function(response) {
        if (response.success) {
            table.ajax.reload();
            Swal.fire(
            translations.deleted,
            response.message,
            'success'
            );
        }
        },
        error: function(xhr) {
        Swal.fire(
            translations.error,
            xhr.responseJSON.message,
            'error'
        );
        }
    });
    }
});
});
